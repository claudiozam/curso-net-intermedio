CREATE DATABASE sistema_reclamos;

CREATE TABLE reclamos (
	id INT NOT NULL AUTO_INCREMENT,
	titulo VARCHAR(255),
	descripcion VARCHAR(3000),
	estado VARCHAR(255),
	fechaAlta DATETIME,
	PRIMARY KEY (id)
);
